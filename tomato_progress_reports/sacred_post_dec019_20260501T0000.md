## Verification command

```python
python -c "
import hashlib, json, os, fnmatch
manifest = json.load(open('.claude/sacred_manifest.json'))
for path, entry in manifest['entries'].items():
    if entry['type'] == 'directory':
        exclusions = entry.get('log_exclusions', [])
        file_hashes = {}
        for dirpath, dirs, files in os.walk(path.rstrip('/')):
            dirs.sort()
            for fn in sorted(files):
                if any(fnmatch.fnmatch(fn, pat) for pat in exclusions): continue
                full = os.path.join(dirpath, fn)
                rel = os.path.relpath(full, path.rstrip('/')).replace(os.sep, '/')
                with open(full, 'rb') as fh:
                    file_hashes[rel] = hashlib.sha256(fh.read()).hexdigest()
        canonical = json.dumps(file_hashes, sort_keys=True, separators=(',', ':'))
        actual = hashlib.sha256(canonical.encode()).hexdigest()
    else:
        with open(path, 'rb') as f: actual = hashlib.sha256(f.read()).hexdigest()
    print(path, 'PASS' if actual == entry['sha256'] else 'FAIL')
"
```
